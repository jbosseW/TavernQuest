```
    // FAR NORTH
    if tileY < -200 then
        return FROSTBOUND_REACH  // ❄️ Northern ice pole

    // FAR WEST
    elseif tileX < -200 then
        if tileY in [0,60] then
            return GREAT_WESTERN_ISLE  // ▓ Distant continent

            return WESTERN_OCEAN  // 🌊 More ocean

            return ASHEN_ARCHIPELAGO  // 🌋 Volcanic islands
        else
            return WESTERN_OCEAN  // 🌊 Ocean between

    elseif tileX < -100 then
        return WESTERN_OCEAN  // 🌊 Outer Waters

    // NORTH
    elseif tileY < 0 then
        return GREAT_ENDLESS_DESERT  // ≈ Northern desert

    // WEST
    elseif tileX < 0 then
        return SCORCHED_SANDS  // ≈ Western desert

    // *** NEW SOUTHERN CHECKS ***

    // EXTREME SOUTH
    elseif tileY >= 250 then
        return SOUTHERN_WASTES  // ≈ Infinite southern desert

    // FAR SOUTH
    elseif tileY >= 150 then
        return SOUTHERN_OCEAN  // 🌊 Sunless Sea

    // SOUTH
    elseif tileY >= 64 and tileY < 150 and tileX >= 0 and tileX < 64 then
        return WASTES_OF_CALIDAR  // ═ Glass desert (bounded now!)

    // FAR EAST
    elseif tileX >= 150 then
        return SHIMMERING_SEA  // 🌊 Infinite east ocean

    // Check bounded regions (main continent, gnomish isles, silver seas)
    // ...

    return SILVER_SEAS  // Default ocean
```
